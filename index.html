import { GoogleGenerativeAI } from "@google/generative-ai"; import express from 'express'; import fs from 'fs'; const app = express(); app.use(express.json()); app.use(express.static('.')); const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY || "SU_API_KEY_AQUI"); app.post('/api/diagnostico', async (req, res) => { try { const { marca, modelo, sintoma, descartes, usuario } = req.body; const tips = fs.readFileSync('./tips.js', 'utf8'); const model = genAI.getGenerativeModel({ model: "gemini-pro" }); const prompt = `Actúa como técnico experto del Método OC. Usa esta info: ${tips}. Caso: ${marca} ${modelo}, falla: ${sintoma}, descartes: ${descartes}. Usuario: ${usuario}. Responde directo al grano.`; const result = await model.generateContent(prompt); res.json({ text: result.response.text() }); } catch (err) { console.error("ERROR DETALLADO:", err); // Esto aparecerá en los logs de Coolify res.status(500).json({ text: "Error en el servidor: " + err.message }); } }); app.listen(process.env.PORT || 3000, '0.0.0.0', () => console.log("🚀 Sistema Método OC activo..."));
